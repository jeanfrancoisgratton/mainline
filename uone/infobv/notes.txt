private bool afficheChangeLog()
        {
            StreamReader sr = null;
            bool bRes = true;
            try
            {
                Assembly _assembly = Assembly.GetExecutingAssembly();
                sr = new StreamReader(_assembly.GetManifestResourceStream("Standby.Resources.CHANGELOG.txt"));
                string sLine;

                while ((sLine = sr.ReadLine()) != null)
                    tbCL.Text += sLine + "\n";
            }

            catch (Exception ex)
            {
                MessageBox.Show(ex.Message);
                bRes = false;
            }

            finally
            {
                if (sr != null) sr.Close();
            }

            return bRes;
        }
